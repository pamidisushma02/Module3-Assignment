import matplotlib.pyplot as plt
import random
import numpy as np
randomlist = [random.randint(1,100) for i in range(0, 5)]  #create a list of 5 random numbers
x = [0,1,2,3,4]

plt.scatter(x, randomlist) #Scatter
print(randomlist)

plt.plot(x, randomlist) #plot the data
plt.xticks(range(0,len(randomlist)+1, 1)) #set the tick frequency on x-axis

plt.ylabel('Random number') #set the label for y axis
plt.xlabel('index') #set the label for x-axis
plt.title("Scatter Plot for 5 Random integers") #set the title of the graph
plt.show() #display the graph
